Public methods: implementation of super-class declarations

    methods
        %------------------------------------------------------------------
        function setDefaultProps(this)
            this.init_value = 0;
            this.slope      = 0;
        end

        %------------------------------------------------------------------
        function val = getValue(this,time)
            val = deal(this.init_value) + this.slope * (time-this.init_time);
        end
    end
